    newtype Cont r a = Cont { runCont :: ((a -> r) -> r) } -- r は計算全体の最終の型
instance Monad (Cont r) where 
    return a       = Cont $ \k -> k a                       
    -- i.e. return a = \k -> k a 
    (Cont c) >>= f = Cont $ \k -> c (\a -> runCont (f a) k) 
    -- i.e. m >>= f = \k -> m (\a -> f a k)

      Cont: {
