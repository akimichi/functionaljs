      flatMap: (parser) => {
    var self = this;
    return (f) => {
      return (input) => {
        return (self.parse(parser)(input)).match({
          empty: () => {
            return List.empty();
          },
          cons: (pair,_) => {
            return pair.match({
              cons: (v, out) => {
                return self.parse(f(v))(out);
              }
            });
