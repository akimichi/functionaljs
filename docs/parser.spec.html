var ParserResult = {
  unit: (value) => {
    return List.empty();
  }
};

var Parser = {
  parse: (parser) => {
    return (input) => {
      return parser(input);
    };
  },
  item: (input) => {
    return List.match(input,{
      empty: (_) => {
        return List.empty();
      },
      cons: (head, tail) => {
        return List.unit(Pair.cons(head, tail));
      }
    });
  }
};
