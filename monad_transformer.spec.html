    describe("MaybeTモナドをテストする",() => {
  it("MaybeT#unitをテストする", (next) => {
    Maybe.match(MaybeT.unit(1)(ID),{
      nothing: (_) => {
        return expect.fail();
      },
      just: (v) => {
        expect(
          v
        ).to.eql(
          1
        );
      }
    });
    next();
  });
  it("MaybeT#flatMapをテストする", (next) => {
    var instanceM = MaybeT.unit(1)(ID);
    var identity = (x) => {
      return x;
    };
    var double = (x) => {
      return 2 * x;
    };
    expect(
      MaybeT.flatMap(instanceM)(double)(ID)
    ).to.eql(
      2
    );
    next();
  });
});
